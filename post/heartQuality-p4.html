JAVASCRIPT

function toggleOptions() {
    var options = document.querySelector('.option');
    options.style.scale = options.style.scale === '1' ? '0' : '1';
}

function selectOption(event) {
    var selectedText = event.target.innerText;
    var selectionBoxLabel = document.querySelector('.selection-box label');
    selectionBoxLabel.innerText = selectedText;

    // Show the corresponding image based on the selected option
    var loveBoxImages = document.querySelectorAll('.love-box img');
    loveBoxImages.forEach(function (img) {
        img.style.scale = img.classList.contains(selectedText) ? '1' : '0';
    });

    toggleOptions();
}
